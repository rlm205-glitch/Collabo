147    def test_verify_email_success(self):
148        # register a user, grab their token, verify it
149        self.client.post(
150            path="/authentication/register/",
151            data=json.dumps({"email": "xaj3@case.edu", "password": "Password123!",
152                             "first_name": "John", "last_name": "Smith"}),
153            content_type="application/json"
154        )
155
156        User = get_user_model()

158        token_rec = EmailVerificationToken.objects.get(user=user)

161        reset_link = mail.outbox[0].body

163
164        response = self.client.post(
165            path="/authentication/verify-email/",
166            data=json.dumps({"token": raw_token}),
167            content_type="application/json"
168        )
169        self.assertEqual(response.status_code, SUCCESS)
170
171        # user should now be active
172        user.refresh_from_db()
173        self.assertTrue(user.is_active)

On database backends with no transaction support, TestCase behaves as TransactionTestCase.
